SELECT COUNT(*)
FROM Users U
GROUP BY U.UserID;

2.

SELECT COUNT(*)
FROM Users U
GROUP BY U.UserID
HAVING U.Location == "New York";

3.

SELECT COUNT(*)
FROM Categories
GROUP BY ItemID
HAVING COUNT(*) == 4;

4.

SELECT ItemID
FROM Items
WHERE Currently IN (SELECT MAX(CURRENTLY) FROM Items);

5.

SELECT COUNT(*)
FROM Users, Items
WHERE Users.UserID == Items.UserID AND 
      Users.Rating > 1000
GROUP BY Users.UserID AND Items.UserID

6.

SELECT COUNT(*)
FROM Items, Bids
WHERE Items.UserID == Bids.UserID
GROUP BY Items.UserID AND Bids.UserID

7.

SELECT COUNT(*)
FROM Categories, Items, Bids
WHERE Items.Category == Categories.Category 
	AND Items.ItemID == Bids.ItemID
	AND Bids.Amount > 100
GROUP BY Categories.Category
